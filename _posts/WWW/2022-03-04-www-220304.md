const h1 = document.querySelector(".hello h1");

function handleTitleClick() {
    const currentColor = h1.style.color;

    if (currentColor === "blue") {

**js**

```js
const h1 = document.querySelector(".hello h1");

function handleTitleClick() {
    const activeClass = "active";
    if (h1.classList.contains(activeClass) { // .contains == calssList의 function
        // class name이 active인지 확인
        h1.classList.remove(activeClass); // .active 제거 && .fontChange 유지
    } else {
        h1.classList.add(activeClass); // .fontChange 유지 && .active 추가
    }
}

h1.addEventListener("click", handleTitleClick);

```
